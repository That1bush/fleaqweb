let allPosts = []; async function fetchPosts() { try { // This looks for the posts.json file in your repo const response = await fetch('posts.json'); if (!response.ok) throw new Error('Could not find posts.json'); allPosts = await response.json(); displayPosts(); } catch (e) { console.error(e); document.getElementById('blog-container').innerText = "Error loading posts. Make sure posts.json exists!"; } } function displayPosts(filter = 'All', search = '') { const container = document.getElementById('blog-container'); container.innerHTML = ''; const filtered = allPosts.filter(p => { const matchesTag = filter === 'All' || p.tags.includes(filter); const matchesSearch = p.title.toLowerCase().includes(search.toLowerCase()); return matchesTag && matchesSearch; }); filtered.forEach((post) => { const card = document.createElement('div'); card.className = 'card'; card.onclick = () => openPost(post); card.innerHTML = `
By ${post.author} | ${post.date}

${post.title}

${post.summary}

${post.tags.map(t => `${t}`).join('')}
`; container.appendChild(card); }); } function openPost(post) { const modal = document.getElementById('post-modal'); document.getElementById('modal-body').innerHTML = `
By ${post.author} • ${post.date}

${post.title}

${post.tags.map(t => `${t}`).join('')}

${post.content}

`; modal.style.display = 'block'; document.body.style.overflow = 'hidden'; } function closePost() { document.getElementById('post-modal').style.display = 'none'; document.body.style.overflow = 'auto'; } function filterTags(tag) { document.querySelectorAll('.tag-btn').forEach(btn => btn.classList.remove('active')); event.target.classList.add('active'); displayPosts(tag, document.getElementById('searchInput').value); } document.getElementById('searchInput').oninput = (e) => displayPosts('All', e.target.value); fetchPosts();
